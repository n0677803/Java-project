CREATE TABLE Customer (
	customerName VARCHAR(20) PRIMARY KEY,
	street VARCHAR(15),
	customerCity VARCHAR(15));

CREATE TABLE Branch (
	branchName VARCHAR(15) PRIMARY KEY,
	branchCity VARCHAR(15),
	Assets DECIMAL(8,2) DEFAULT 0);

CREATE TABLE Loan (
	customerName VARCHAR(20) CONSTRAINT LnCusNmFK REFERENCES Customer ON DELETE CASCADE,
	branchName VARCHAR(15) CONSTRAINT LnBrchNmFK REFERENCES Branch ON DELETE CASCADE,
	loanNumber INT PRIMARY KEY,
	amount DECIMAL(8,2) DEFAULT 0);

CREATE TABLE Deposit (
	customerName VARCHAR(20) CONSTRAINT DepCusNmFK REFERENCES Customer ON DELETE CASCADE,
	branchName VARCHAR(15) CONSTRAINT DepBrchNmFK REFERENCES Branch ON DELETE CASCADE,
	accountNumber INT PRIMARY KEY,
	balance DECIMAL(8,2) DEFAULT 0);
